

### 顺序表

用**顺序存储**的方式实现线性表的顺序存储。 

把逻辑上相邻的元素存储在物理位置上也相邻的存储单元中，元素之间的关系由存储单元的邻接关系来体现

```c
#include <stdio.h>
#include <malloc.h>

struct Array {
	int length = 0;
	int capacity = 32;
	int* _items = (int *) malloc(sizeof(int)*3);
};

void SetElement(Array& arr, int index, int value) {
	if (arr.capacity <= index) {
		printf("index error");
		return;
	}
	int* p = arr._items + index;
	*p = value;
}

int GetElement(Array& arr, int index) {
	if (arr.length < index) {
		throw("index error");
	}
	return *(arr._items + index);
}

int main(int argc, char const* argv[])
{
	Array arr;
	for (int i = 0; i < 3; i++) {
		SetElement(arr, i, i);
	}
	for (int i = 0; i < 3; ++i)
	{
		int j = GetElement(arr, i);
		printf("%d\n", j);
	}
	free(arr._items);
	return 0;
}
```
